        id INT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
        date_start DATE,
        date_end DATE,
        total_price FLOAT,
        id_user INT UNSIGNED,
        id_car INT UNSIGNED,
        CONSTRAINT fk_id_user FOREIGN KEY (id_user) REFERENCES user (id),

    );
